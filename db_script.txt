DROP TABLE IF EXISTS users CASCADE;
DROP TABLE IF EXISTS bugs CASCADE;
DROP TABLE IF EXISTS changes;

CREATE TABLE users (
	user_id INT GENERATED ALWAYS AS IDENTITY,
	username TEXT UNIQUE NOT NULL,
	firstlastname TEXT,
	user_password TEXT NOT NULL,
	PRIMARY KEY(user_id)
);

CREATE TABLE bugs (
	bug_id INT GENERATED ALWAYS AS IDENTITY,
	bug_title TEXT NOT NULL,
	bug_details TEXT,
	creator_id INT,
	bug_status TEXT NOT NULL,
	bug_urgency TEXT NOT NULL,
	bug_crit TEXT NOT NULL,
	bug_whenfound DATE,
	
	PRIMARY KEY(bug_id),
    CONSTRAINT fk_user
      FOREIGN KEY(creator_id) 
	  REFERENCES users(user_id)
);

CREATE TABLE changes(
	change_id INT GENERATED ALWAYS AS IDENTITY,
	bug_id INT,
	changer_id INT,
	change_date DATE,
	change_action TEXT,
	change_comment TEXT,
	
	PRIMARY KEY(change_id),
    CONSTRAINT fk_user
      FOREIGN KEY(changer_id) 
	  REFERENCES users(user_id),
	CONSTRAINT fk_bug
      FOREIGN KEY(bug_id) 
	  REFERENCES bugs(bug_id)
);
